编写一个函数解析Query String并返回解析后的对象。
```js
// url = www.123.com?a=1&b=2
function match(url){
  let querys = url.split('?')[1].split('&')
  let result = {}
  querys.forEach(item=>{
    let [key,value] = item.split('=')
    result[key]=value
  })
  return result
}
match('url = www.123.com?a=1&b=2')
```
